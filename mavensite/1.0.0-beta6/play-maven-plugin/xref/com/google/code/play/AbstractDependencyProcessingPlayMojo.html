102     protected Artifact getDependencyArtifact( Set<?> classPathArtifacts, String groupId, String artifactId, String type )
103     {
104         Artifact result = null;
105         for ( Iterator<?> iter = classPathArtifacts.iterator(); iter.hasNext(); )
106         {
107             Artifact artifact = (Artifact) iter.next();
108             if ( groupId.equals( artifact.getGroupId() ) && artifactId.equals( artifact.getArtifactId() )
109                 && type.equals( artifact.getType() ) )
110             {
111                 result = artifact;
112                 break;
113             }
114         }
115         return result;
116     }

118     protected Set<Artifact> getDependencyArtifacts( Set<?> classPathArtifacts, Artifact rootArtifact )
119         throws DependencyTreeBuilderException
120     {
121         Set<Artifact> result = null;
122 
123         buildDependencyTree();
124         DependencyNode artifactNode = findArtifactNode( rootArtifact, rootNode );
125         if ( artifactNode != null )
126         {
127             result = new HashSet<Artifact>();
128             addDependencyArtifacts( result, classPathArtifacts, artifactNode );
129         }
130         else
131         {
132             result = Collections.emptySet();
133         }
134 
135         return result;
136     }

138     private Set<Artifact> getDependencyArtifactsExt( Set<?> classPathArtifacts, Artifact rootArtifact )
139         throws DependencyTreeBuilderException
140     {
141         Set<Artifact> result = null;
142 
143         buildDependencyTree();
144         DependencyNode artifactNode = findArtifactNode( rootArtifact, rootNode );
145         if ( artifactNode != null )
146         {
147             result = new HashSet<Artifact>();
148             addDependencyArtifacts( result, classPathArtifacts, artifactNode );
149             if ( result.isEmpty() )
150             {
151                 Set<?> projectArtifacts = project.getArtifacts();
152                 for ( Iterator<?> iter = projectArtifacts.iterator(); iter.hasNext(); )
153                 {
154                     Artifact artifact = (Artifact) iter.next();
155                     if ( artifact != rootArtifact && artifact.getGroupId().equals( rootArtifact.getGroupId() )
156                         && artifact.getArtifactId().equals( rootArtifact.getArtifactId() ) )
157                     {
158                         Set<Artifact> tmp = getDependencyArtifacts( classPathArtifacts, artifact );
159                         result.addAll( tmp );
160                     }
161                 }
162             }
163         }
164         else
165         {
166             result = Collections.emptySet();
167         }
168 
169         return result;
170     }

172     private void addDependencyArtifacts( Set<Artifact> collection, Set<?> classPathArtifacts, DependencyNode artifactNode )
173     {
174         if ( artifactNode.getState() == DependencyNode.INCLUDED )
175         {
176             Artifact artifact = artifactNode.getArtifact();
177             // don't use this artifact, because it can be unresolved
178             // find corresponding artifact in "classPathArtifacts" set
179             // (only if exists, if does not exist - we don't need it)
180             for ( Iterator<?> iter = classPathArtifacts.iterator(); iter.hasNext(); )
181             {
182                 Artifact a = (Artifact) iter.next();
183                 if ( areArtifactsEqual( artifact, a ) )
184                 {
185                     collection.add( a );
186                     break;
187                 }
188             }
189             List<?> childDependencyNodes = artifactNode.getChildren();
190             for ( Iterator<?> iter = childDependencyNodes.iterator(); iter.hasNext(); )
191             {
192                 DependencyNode childNode = (DependencyNode) iter.next();
193                 addDependencyArtifacts( collection, classPathArtifacts, childNode );
194             }
195         }
196     }

198     private DependencyNode findArtifactNode( Artifact artifact, DependencyNode findRootNode )
199     {
200         DependencyNode result = null;
201         if ( findRootNode.getState() == DependencyNode.INCLUDED )
202         {
203             if ( findRootNode.getArtifact().equals( artifact ) )
204             {
205                 result = findRootNode;
206             }
207             else
208             {
209                 List<?> childDependencyNodes = findRootNode.getChildren();
210                 for ( Iterator<?> iter = childDependencyNodes.iterator(); iter.hasNext(); )
211                 {
212                     DependencyNode childNode = (DependencyNode) iter.next();
213                     DependencyNode tmp = findArtifactNode( artifact, childNode );
214                     if ( tmp != null )
215                     {
216                         result = tmp;
217                         break;
218                     }
219                 }
220             }
221         }
222         return result;
223     }

225     private boolean areArtifactsEqual( Artifact a1, Artifact a2 )
226     {
227         boolean result =
228             a1.getGroupId().equals( a2.getGroupId() )
229                 && a1.getArtifactId().equals( a2.getArtifactId() )
230                 && a1.getType().equals( a2.getType() )
231                 && ( a1.getClassifier() == null ? a2.getClassifier() == null
232                                 : a1.getClassifier().equals( a2.getClassifier() ) );
233         return result;
234     }

255     protected Set<String> getProvidedModuleNames( ConfigurationParser configParser, String playId, boolean forceProdMode )
256         throws IOException
257     {
258         Map<String, String> modulePaths = configParser.getModules();
259 
260         Set<String> moduleNames = new HashSet<String>();
261         for ( String moduleName : modulePaths.keySet() )
262         {
263             moduleNames.add( moduleName );
264             // getLog().debug( String.format( "Module: %s", moduleName ) );
265         }
266         //TODO-LATER
267         //String modeStr = configParser.getProperty( "application.mode" );
268         ////TODO-validate
269         //if (!forceProdMode && !modeStr.equalsIgnoreCase( "prod" ))
270         //{
271         //    moduleNames.add( "docviewer" );//TODO-add only in dev mode
272         //}
273         if ( !forceProdMode )
274         {
275             moduleNames.add( "docviewer" );
276         }
277         if ( isTestProfile( playId ) )
278         {
279             moduleNames.add( "testrunner" );
280         }
281 
282         return moduleNames;
283     }

285     private boolean isTestProfile( String playId )
286     {
287         boolean result = false;
288         if ( playId != null )
289         {
290             result = "test".equals( playId ) || ( playId.startsWith( "test-" ) && playId.length() > 5 );
291         }
292         return result;
293     }
